

-- 安全管护_总览 --
DROP TABLE IF EXISTS "dm_security"."dm_security_overview";
CREATE TABLE IF NOT EXISTS "dm_security"."dm_security_overview"
(
    "id" serial,
    "s_sum" varchar(100),
    "now_num" varchar(100),
    "mp_num" varchar(100),
    "report_num" varchar(100),
    "reform_num" varchar(100),
    "reform_perent" varchar(100),
    "company" varchar(100),
    "create_time" timestamp(6)
)WITH(orientation=row,compression=no) ;
COMMENT ON TABLE "dm_security"."dm_security_overview" IS '安全管护_总览';
COMMENT ON COLUMN dm_security.dm_security_overview.id IS '系统id';
COMMENT ON COLUMN dm_security.dm_security_overview.s_sum IS '隐患总数';
COMMENT ON COLUMN dm_security.dm_security_overview.now_num IS '现有隐患总数';
COMMENT ON COLUMN dm_security.dm_security_overview.mp_num IS '管护系统来源数';
COMMENT ON COLUMN dm_security.dm_security_overview.report_num IS '隐患报告来源数';
COMMENT ON COLUMN dm_security.dm_security_overview.reform_num IS '整改数';
COMMENT ON COLUMN dm_security.dm_security_overview.reform_perent IS '整改率';
COMMENT ON COLUMN dm_security.dm_security_overview.company IS '所属公司';
COMMENT ON COLUMN dm_security.dm_security_overview.create_time IS '创建时间';
