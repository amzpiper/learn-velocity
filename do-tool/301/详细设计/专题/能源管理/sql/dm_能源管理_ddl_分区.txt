

-- 能源管理_总览 --
DROP TABLE IF EXISTS "dm_energy"."dm_energy_overview";
CREATE TABLE IF NOT EXISTS "dm_energy"."dm_energy_overview"
(
    "id" serial,
    "electric" varchar(100),
    "e_year" varchar(100),
    "e_ring" varchar(100),
    "water" varchar(100),
    "w_year" varchar(100),
    "w_ring" varchar(100),
    "heat" varchar(100),
    "h_year" varchar(100),
    "h_ring" varchar(100),
    "date_time" timestamp(6),
    "company" varchar(100),
    "create_time" timestamp(6)
)WITH(orientation=row,compression=no) ;
COMMENT ON TABLE "dm_energy"."dm_energy_overview" IS '能源管理_总览';
COMMENT ON COLUMN dm_energy.dm_energy_overview.id IS '系统id';
COMMENT ON COLUMN dm_energy.dm_energy_overview.electric IS '电能能耗';
COMMENT ON COLUMN dm_energy.dm_energy_overview.e_year IS '电能同比';
COMMENT ON COLUMN dm_energy.dm_energy_overview.e_ring IS '电能环比';
COMMENT ON COLUMN dm_energy.dm_energy_overview.water IS '水能能耗';
COMMENT ON COLUMN dm_energy.dm_energy_overview.w_year IS '水能同比';
COMMENT ON COLUMN dm_energy.dm_energy_overview.w_ring IS '水能环比';
COMMENT ON COLUMN dm_energy.dm_energy_overview.heat IS '热能能耗';
COMMENT ON COLUMN dm_energy.dm_energy_overview.h_year IS '热能同比';
COMMENT ON COLUMN dm_energy.dm_energy_overview.h_ring IS '热能环比';
COMMENT ON COLUMN dm_energy.dm_energy_overview.date_time IS '刻度时间';
COMMENT ON COLUMN dm_energy.dm_energy_overview.company IS '所属公司';
COMMENT ON COLUMN dm_energy.dm_energy_overview.create_time IS '创建时间';
