

-- 智能运维_总览 --
DROP TABLE IF EXISTS "dm_operation"."dm_operation_overview";
CREATE TABLE IF NOT EXISTS "dm_operation"."dm_operation_overview"
(
    "id" serial,
    "device_sum" varchar(100),
    "alarm_num" varchar(100),
    "handle_num" varchar(100),
    "run_num" varchar(100),
    "stop_num" varchar(100),
    "company" varchar(100),
    "create_time" timestamp(6)
)WITH(orientation=row,compression=no) ;
COMMENT ON TABLE "dm_operation"."dm_operation_overview" IS '智能运维_总览';
COMMENT ON COLUMN dm_operation.dm_operation_overview.id IS '系统id';
COMMENT ON COLUMN dm_operation.dm_operation_overview.device_sum IS '设备总数';
COMMENT ON COLUMN dm_operation.dm_operation_overview.alarm_num IS '告警数';
COMMENT ON COLUMN dm_operation.dm_operation_overview.handle_num IS '处理数';
COMMENT ON COLUMN dm_operation.dm_operation_overview.run_num IS '运行数';
COMMENT ON COLUMN dm_operation.dm_operation_overview.stop_num IS '停机数';
COMMENT ON COLUMN dm_operation.dm_operation_overview.company IS '所属公司';
COMMENT ON COLUMN dm_operation.dm_operation_overview.create_time IS '创建时间';
