

-- 产业运营保障_资产租赁 --
DROP TABLE IF EXISTS "dm_assetoperation"."dm_assetoperation_assetlease";
CREATE TABLE IF NOT EXISTS "dm_assetoperation"."dm_assetoperation_assetlease"
(
    "id" serial,
    "attr_name" varchar(100),
    "value" varchar(100),
    "company" varchar(100),
    "create_time" timestamp(6)
)WITH(orientation=row,compression=no) ;
COMMENT ON TABLE "dm_assetoperation"."dm_assetoperation_assetlease" IS '产业运营保障_资产租赁';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_assetlease.id IS '系统id';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_assetlease.attr_name IS '属性名称';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_assetlease.value IS '属性值';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_assetlease.company IS '所属公司';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_assetlease.create_time IS '创建时间';


-- 产业运营保障_服务 --
DROP TABLE IF EXISTS "dm_assetoperation"."dm_assetoperation_service";
CREATE TABLE IF NOT EXISTS "dm_assetoperation"."dm_assetoperation_service"
(
    "id" serial,
    "attr_name" varchar(100),
    "gmv_year" varchar(100),
    "gmv_ring" varchar(100),
    "order_year" varchar(100),
    "order_ring" varchar(100),
    "service_type" varchar(100),
    "revenue_year" varchar(100),
    "revenue_ring" varchar(100),
    "park_year" varchar(100),
    "park_ring" varchar(100),
    "date_time" timestamp(6),
    "company" varchar(100),
    "create_time" timestamp(6)
)WITH(orientation=row,compression=no) ;
COMMENT ON TABLE "dm_assetoperation"."dm_assetoperation_service" IS '产业运营保障_服务';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.id IS '系统id';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.attr_name IS '注册用户数';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.gmv_year IS 'GMV成交同比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.gmv_ring IS 'GMV成交环比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.order_year IS '订单量同比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.order_ring IS '订单量环比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.service_type IS '服务类型';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.revenue_year IS '当月营收同比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.revenue_ring IS '当月营收环比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.park_year IS '停车效率同比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.park_ring IS '停车效率环比';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.date_time IS '刻度时间';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.company IS '所属公司';
COMMENT ON COLUMN dm_assetoperation.dm_assetoperation_service.create_time IS '创建时间';
