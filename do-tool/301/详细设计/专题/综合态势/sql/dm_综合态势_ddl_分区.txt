

-- 综合态势_总览 --
DROP TABLE IF EXISTS "dm_comprehensive"."dm_comprehensive_overview";
CREATE TABLE IF NOT EXISTS "dm_comprehensive"."dm_comprehensive_overview"
(
    "id" serial,
    "area" varchar(100),
    "department_num" varchar(100),
    "visits_num" varchar(100),
    "bed_num" varchar(100),
    "service_num" varchar(100),
    "park_num" varchar(100),
    "company" varchar(100),
    "create_time" timestamp(6)
)WITH(orientation=row,compression=no) ;
COMMENT ON TABLE "dm_comprehensive"."dm_comprehensive_overview" IS '综合态势_总览';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.id IS '系统id';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.area IS '建筑面积';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.department_num IS '科室数量';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.visits_num IS '日就诊量';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.bed_num IS '床位数';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.service_num IS '服务人员数';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.park_num IS '车位数';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.company IS '所属公司';
COMMENT ON COLUMN dm_comprehensive.dm_comprehensive_overview.create_time IS '创建时间';
